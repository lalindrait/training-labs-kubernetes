

2.1 - Deploy your first pod
-------------------------

apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-pod
spec:
  selector:
    matchLabels:
      app: nginx
  replicas: 2 # tells deployment to run 2 pods matching the template
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:latest
        ports:
        - containerPort: 80




kubectl apply -f first-nginx-pod.yaml
kubectl get pods -o wide
kubectl get pods -o yaml

kubectl describe pods first-nginx

kubectl logs first-nginx
kubectl logs multipod --container syncer

kubectl exec first-nginx -- hostname
kubectl exec -it first-nginx -- bash

kubectl delete pods first-nginx-pod
