3.1 - Running your first deployment
---------------------------------


apiVersion: apps/v1
kind: Deployment
metadata:
  name: lv-nginx
spec:
  replicas: 3
  selector:
    matchLabels:
      app: nginx
  revisionHistoryLimit: 5
  progressDeadlineSeconds: 300
  minReadySeconds: 10
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxUnavailable: 1
      maxSurge: 1
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: lv-nginx
        image: lalindrait/lvrocky91:latest
        ports:
        - containerPort: 8080
        resources:
          requests:
            memory: "256M"
            cpu: "1000m"
          limits:
            memory: "512M"
            cpu: "2000m"




kubectl apply -f first-deployment.yml
kubectl get all -o wide

kubectl get deploy
kubectl get deploy -o wide
kubectl get deploy lv-nginx

kubectl describe deploy lv-nginx

kubectl get rs
kubectl get rs -o wide


kubectl describe rs lv-nginx-847675d7f7
